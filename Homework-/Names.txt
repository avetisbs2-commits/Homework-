package inheritance_homework;

public class Car {
    int passengerCount;
    char engineType;

    Car(int passengerCount, char engineType) {
        setPassengerCount(passengerCount);
        setEngineType(engineType);
    }

    int getPassengerCount(){

        return passengerCount;
    }

    void setPassengerCount(int passengerCount){
        if (passengerCount >= 2) {
            this.passengerCount = passengerCount;
        }
    }

    char getEngineType(){
        return engineType;
    }

    void setEngineType(char engineType){
        if (Character.isLetter(engineType)) {
            this.engineType = engineType;
        }
    }
}


class Truck extends Car {
    int capacity;

    Truck(int passengerCount, char engineType) {
        super(passengerCount, engineType);
    }

    int getCapacity() {
        return capacity;
    }

    void setCapacity(int capacity) {
        if (capacity >= 0) {
            this.capacity = capacity;
        }
    }

    public static void main(String[] args) {
        Truck truck = new Truck(6, '1');

        System.out.println("Passenger Count: " + truck.getPassengerCount());
        System.out.println("Engine Type: " + truck.getEngineType());
        truck.setCapacity(12);
        System.out.println("Capacity : " + truck.getCapacity());

    }
}
